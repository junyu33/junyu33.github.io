def is_surrounded(board, color, x, y):
    # Directions for checking neighbors
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    visited = set()
    to_check = [(x, y)]
    while to_check:
        cx, cy = to_check.pop()
        if (cx, cy) in visited:
            continue
        visited.add((cx, cy))
        for dx, dy in directions:
            nx, ny = cx + dx, cy + dy
            if 0 <= nx < 9 and 0 <= ny < 9 and (nx, ny) not in visited:

    return True
